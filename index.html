📊 Data Analyst nima qiladi?

SQL Cheatsheet

Asosiy SELECT
SELECT col1, col2, col3
FROM table_name
WHERE condition = 'value'
  AND amount > 100
ORDER BY col1 DESC
LIMIT 100;
Guruhlash
SELECT
  category,
  COUNT(*) AS cnt,
  SUM(amount) AS total,
  AVG(amount) AS avg_val,
  MAX(amount) AS max_val
FROM sales
GROUP BY category
HAVING COUNT(*) > 5
ORDER BY total DESC;
JOIN
SELECT
  o.order_id,
  c.name,
  p.product,
  o.amount
FROM orders o
LEFT JOIN customers c
  ON o.customer_id = c.id
INNER JOIN products p
  ON o.product_id = p.id
WHERE o.date >= '2024-01-01';
Sana bo'yicha
SELECT
  DATE_TRUNC('month', date) AS month,
  COUNT(*) AS orders,
  SUM(revenue) AS revenue
FROM orders
WHERE date BETWEEN
  '2024-01-01' AND '2024-12-31'
GROUP BY 1
ORDER BY 1;
Window Functions
SELECT
  name, salary, dept,
  RANK() OVER (
    PARTITION BY dept
    ORDER BY salary DESC
  ) AS rank,
  SUM(salary) OVER (
    PARTITION BY dept
  ) AS dept_total,
  LAG(salary) OVER(
    ORDER BY hire_date
  ) AS prev_sal
FROM employees;
Foiz Hisoblash
SELECT
  category,
  COUNT(*) AS cnt,
  ROUND(
    COUNT(*) * 100.0 /
    SUM(COUNT(*)) OVER(), 2
  ) AS pct
FROM orders
GROUP BY category
ORDER BY cnt DESC;
